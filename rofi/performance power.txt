git clone https://github.com/AdnanHodzic/auto-cpufreq.git
cd auto-cpufreq
sudo ./auto-cpufreq-installer --install
sudo auto-cpufreq --install


sudo bash -c 'cat <<EOF > /etc/auto-cpufreq.conf
[charger]
governor = performance
turbo = auto

[battery]
governor = powersave
turbo = never
EOF'


echo "$USER ALL=(ALL) NOPASSWD: /usr/local/bin/auto-cpufreq" | sudo tee /etc/sudoers.d/auto-cpufreq-rofi


mkdir -p ~/.local/bin
cat <<EOF > ~/.local/bin/power-menu.sh
#!/bin/bash
status=\$(auto-cpufreq --stats | grep "CPU usage" -B 1 | head -n 1)
options="Performance\nPower Save\nReset (Auto Mode)"
choice=\$(echo -e "\$options" | rofi -dmenu -p "CPU: \$status")

case "\$choice" in
    Performance) sudo auto-cpufreq --force=performance ;;
    "Power Save") sudo auto-cpufreq --force=powersave ;;
    "Reset (Auto Mode)") sudo auto-cpufreq --force=reset ;;
esac
EOF
chmod +x ~/.local/bin/power-menu.sh


sudo systemctl restart auto-cpufreq
auto-cpufreq --stats
