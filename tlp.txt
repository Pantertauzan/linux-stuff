apply with:
sudo tlp start

sudo tlp-stat -b

# ------------------------------------------------------------------------------
# TLP Configuration: ThinkPad L14 Gen 1 Intel
# ------------------------------------------------------------------------------

# Enable TLP
TLP_ENABLE=1

# --- CPU Settings (Intel Optimized) ---
# Schedutil is great for battery; Performance for AC.
CPU_SCALING_GOVERNOR_ON_BAT=powersave
CPU_SCALING_GOVERNOR_ON_AC=performance

# Energy Performance Preference (EPP)
CPU_ENERGY_PERF_POLICY_ON_BAT=power
CPU_ENERGY_PERF_POLICY_ON_AC=performance

# Performance Limits (30% cap on battery to prevent drain)
CPU_MIN_PERF_ON_BAT=0
CPU_MAX_PERF_ON_BAT=30
CPU_MIN_PERF_ON_AC=0
CPU_MAX_PERF_ON_AC=100

# Turbo Boost
CPU_BOOST_ON_BAT=0
CPU_BOOST_ON_AC=1

# Idle Governor
CPU_IDLE_GOVERNOR_ON_BAT=menu
CPU_IDLE_GOVERNOR_ON_AC=menu

# --- Intel Graphics (i915) ---
# Limit GPU clock speed on battery to reduce heat and power draw
INTEL_GPU_MIN_FREQ_ON_BAT=300
INTEL_GPU_MAX_FREQ_ON_BAT=800
INTEL_GPU_BOOST_FREQ_ON_BAT=800
# Leave MAX/BOOST at default for AC

# --- Platform Profile ---
# Controls the system's power/thermal firmware limits
PLATFORM_PROFILE_ON_BAT=low-power
PLATFORM_PROFILE_ON_AC=performance

# --- Storage & NVMe ---
# Aggressive power saving for the internal NVMe SSD
DEVICES_NVME_POWER_ON_BAT=power_save
DEVICES_NVME_POWER_ON_AC=default
AHCI_RUNTIME_PM_ON_BAT=auto

# Chipset Link Power Management
SATA_LINKPWR_ON_BAT=med_power_with_dipm
SATA_LINKPWR_ON_AC=max_performance

# --- Runtime Power Management ---
# Automatically suspends idle devices
RUNTIME_PM_ON_BAT=auto
RUNTIME_PM_ON_AC=on

# --- Networking (Wi-Fi & Bluetooth) ---
WIFI_PWR_ON_BAT=on
WIFI_PWR_ON_AC=off
WIFI_PWR_LEVEL_ON_BAT=auto

# Disable Bluetooth on startup to save power (turn on manually when needed)
DEVICES_TO_DISABLE_ON_STARTUP="bluetooth"

# --- Audio Power Saving ---
SOUND_POWER_SAVE_ON_BAT=1
SOUND_POWER_SAVE_ON_AC=0
SOUND_POWER_SAVE_CONTROLLER=Y

# --- PCIe & ASPM ---
PCIE_ASPM_ON_BAT=powersave
PCIE_ASPM_ON_AC=performance

# --- USB Power Saving ---
USB_AUTOSUSPEND=1
USB_EXCLUDE_PHONE=1

# --- ThinkPad Battery Care ---
# Longevity thresholds: Start charging at 40%, stop at 80%
START_CHARGE_THRESH_BAT0=40
STOP_CHARGE_THRESH_BAT0=80

# --- Thermal Management ---
# Allow TLP to work alongside thermald for better Intel thermal scaling
TLP_THERMAL_CONTROL=1
